	}
	
4. build.bat
	@echo off - отключает вывод исполняемых команд, 
		@ скрывает саму команду
	chcp 65001 >nul - меняет кодовую страницу на utf-8
		для отображения кириллицы
		и перенаправляет вывод команды в пустоту
	/Fo:build\windows\ — задаёт папку для объектных файлов (.obj)

	/Fe:"build\windows\practice1.exe" — 
		задаёт имя и путь выходного исполняемого файла.
	
	//реализовывать сборку нужно в Developer Command Prompt,
	а не в обычной консоли, она не видит "cl"
	
5. 
	в меню пуск ищем "изменение системных переменных среды"
	во вкладке "дополнительно" -> "переменные среды" 
	в системных переменных изменяем path и добавляем новый путь
	папки с нашим exe. везде "ок" и перезапускаем command prompt
	просто вызываем practice1
	
6.
	открываем все заново и удаляем то, что добавили
	
7. 
	в командной строке
	cd *папка с main.c внутри*
	git config --global user.name "Ivan Ivanov"
	git config --global user.email "ivanov@example.com"
		//задаем имя и емейл если раньше этого не делали.
		они никак не проверяются
	git init - инициализируем
	git status - проверка состояния
	echo build/ > .gitignore
	echo *.obj >> .gitignore
	echo *.exe >> .gitignore
		//создаем и закидываем ненужные для коммитов файлы в гитигнор
	git add build.bat main.c .gitignore - закидываем новые/измененные файлы
	git commit -m "added main.c and buld.bat" - наш коммит
	git log - проверка, если нужно
	
8. 
	измененный main.c
	билдим через бат, чтобы сохранить изменения, запускаем сам exe
	если просто запустили выдаст имя которое в винде, если приписали рядом
	с вызовом exe что-то еще то выдаст это
	
9.
	git status
